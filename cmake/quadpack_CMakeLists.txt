cmake_minimum_required(VERSION 3.24)
project(quadpack Fortran)

add_library(quadpack STATIC
    src/quadpack_single.F90
    src/quadpack_double.F90
    src/quadpack.F90
)
target_compile_definitions(quadpack PRIVATE NOQUAD)
target_include_directories(quadpack PRIVATE src)
set_target_properties(quadpack PROPERTIES
    Fortran_MODULE_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/modules
)
target_include_directories(quadpack PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}/modules>
)
